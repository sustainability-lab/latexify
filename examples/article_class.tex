\section{Code}

\begin{verbatim}
import latexify
latexify.latexify(columns=1)  # single
latexify.latexify(columns=2)  # double
plt.plot(x, y)
latexify.format_axes(plt.gca())
latexify.save_fig('plot.pdf')
\end{verbatim}
